module tb;
  reg  [7:0] d;
  reg  [2:0] s;
  wire y;

  mux8to1_behavioral uut (.d(d), .s(s), .y(y));

  initial begin
    d = 8'b1101_0101;      // Example test vector

    for(int i = 0; i < 8; i = i + 1) begin
      s = i[2:0];
      #5 $display("s = %0b  ->  y = %b  (expected = %b)", s, y, d[i]);
    end

    $finish;
  end
endmodule
