// 4-bit Shift Register 
module shift_register (
    input  wire clk,
    input  wire rst,
input  wire shift_en,
    input  wire d_in,    
    output reg  [3:0] q  
);

    always @(posedge clk or posedge rst) begin
        if (rst)
            q <= 4'b0000;                     
        else if (shift_en)
            q <= {d_in, q[3:1]};             
    end

endmodule
